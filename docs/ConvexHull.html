# Rectangle [0,4] × [0,2] with 15 extreme points
reconstruct_rectangle_calculation <- function() {
  cat("=== CONVEX HULL AREA ESTIMATE FOR RECTANGLE [0,4] × [0,2] ===\n\n")
  
  # The correct setup from our original work:
  cat("Original problem:\n")
  cat("- Region K: Rectangle [0,4] × [0,2]\n")
  cat("- True area of K: 4 × 2 = 8\n")
  cat("- Sample size: n = 100\n")
  cat("- Observed extreme points: V_n = 15\n\n")
  
  # Generate example with rectangle
  set.seed(123)
  n <- 100
  d <- 2
  
  # Generate uniform points in rectangle [0,4] × [0,2]
  points <- matrix(0, nrow = n, ncol = d)
  points[,1] <- runif(n, 0, 4)  # x coordinates in [0,4]
  points[,2] <- runif(n, 0, 2)  # y coordinates in [0,2]
  
  cat("Generated", n, "uniform points in rectangle [0,4] × [0,2]\n")
  
  # Find convex hull
  hull_indices <- chull(points[,1], points[,2])
  hull_vertices <- points[hull_indices, ]
  V_n <- length(hull_indices)
  
  cat("- Number of extreme points V_n =", V_n, "\n")
  
  # Compute convex hull area using shoelace formula
  n_hull <- nrow(hull_vertices)
  area <- 0
  for(i in 1:n_hull) {
    j <- ifelse(i == n_hull, 1, i + 1)
    area <- area + (hull_vertices[i,1] * hull_vertices[j,2] - hull_vertices[j,1] * hull_vertices[i,2])
  }
  vol_hull <- abs(area) / 2
  
  cat("- Convex hull area =", format(vol_hull, digits = 6), "\n\n")
  
  # Apply the volume estimator formula
  vol_hat <- vol_hull / (1 - V_n/n)
  
  cat("Volume estimation:\n")
  cat("- Formula: vol_hat = vol_hull / (1 - V_n/n)\n")
  cat("- Calculation: vol_hat =", format(vol_hull, digits = 6), "/ (1 -", V_n, "/", n, ")\n")
  cat("                      =", format(vol_hull, digits = 6), "/", format(1 - V_n/n, digits = 4), "\n")
  cat("                      =", format(vol_hat, digits = 6), "\n\n")
  
  # Compare with true area
  true_area <- 8
  rel_error <- abs(vol_hat - true_area) / true_area
  
  cat("Results:\n")
  cat("- True rectangle area:", true_area, "\n")
  cat("- Estimated area:", format(vol_hat, digits = 6), "\n")
  cat("- Relative error:", round(rel_error, 4), "\n")
  cat("- Error percentage:", round(rel_error * 100, 2), "%\n\n")
  
  # Show the hull vertices
  cat("Convex hull vertices:\n")
  print(round(hull_vertices, 3))
  
  return(list(
    true_area = true_area,
    hull_area = vol_hull,
    V_n = V_n,
    estimated_area = vol_hat,
    rel_error = rel_error
  ))
}

# Show what the convex hull typically looks like for rectangle
analyze_rectangle_hull <- function() {
  cat("\n=== ANALYSIS: CONVEX HULL OF RECTANGLE ===\n\n")
  
  cat("For uniform points in rectangle [0,4] × [0,2]:\n")
  cat("- Most points will be interior\n")
  cat("- Extreme points will be near the boundary\n")
  cat("- Convex hull will approximate the rectangle\n")
  cat("- With enough points, hull area → rectangle area (8)\n\n")
  
  # Show typical behavior
  n_trials <- 10
  results <- matrix(0, nrow = n_trials, ncol = 4)
  colnames(results) <- c("V_n", "Hull_Area", "Estimated_Area", "Error")
  
  for(i in 1:n_trials) {
    # Generate points in rectangle
    n <- 100
    points <- matrix(0, nrow = n, ncol = 2)
    points[,1] <- runif(n, 0, 4)
    points[,2] <- runif(n, 0, 2)
    
    # Get hull
    hull_indices <- chull(points[,1], points[,2])
    hull_vertices <- points[hull_indices, ]
    V_n <- length(hull_indices)
    
    # Compute area
    n_hull <- nrow(hull_vertices)
    area <- 0
    for(j in 1:n_hull) {
      k <- ifelse(j == n_hull, 1, j + 1)
      area <- area + (hull_vertices[j,1] * hull_vertices[k,2] - hull_vertices[k,1] * hull_vertices[j,2])
    }
    vol_hull <- abs(area) / 2
    
    # Estimate
    vol_hat <- vol_hull / (1 - V_n/n)
    error <- abs(vol_hat - 8) / 8
    
    results[i, ] <- c(V_n, vol_hull, vol_hat, error)
  }
  
  cat("Results from", n_trials, "trials:\n")
  print(round(results, 4))
  
  cat("\nSummary:\n")
  cat("- Mean V_n:", round(mean(results[,1]), 1), "\n")
  cat("- Mean hull area:", round(mean(results[,2]), 3), "\n")
  cat("- Mean estimated area:", round(mean(results[,3]), 3), "\n")
  cat("- Mean relative error:", round(mean(results[,4]), 4), "\n")
}

# Run the analysis
result <- reconstruct_rectangle_calculation()
